## 테이블 설계
EXPEND: 지출 본체 테이블 <br>
EXPEND_CLIENT: EXPEND 와 CLIENT 간 다대다(N:M) 관계를 풀어내는 테이블 <br>
CLIENT: 거래처 정보 테이블 <br>
EXPEND_SUBJECT: EXPEND 의 지출 항목 테이블 <br>
ACCOUNT_SUBJECT: EXPEND_SUBJECT 가 어떤 회계 계정(ACCOUNT_SUBJECT)에 속하는지를 나타내도록 구성 <br>

## DDL
```sql
CREATE TABLE EXPEND ( 
    id BIGINT PRIMARY KEY AUTO_INCREMENT, 
    expend_date DATE NOT NULL COMMENT '지출 날짜', 
    total_amount DECIMAL(18, 2) NOT NULL COMMENT '총 지출 금액', 
    description VARCHAR(255) COMMENT '지출 설명', 
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '생성 일시',
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정 일시'
); 


CREATE TABLE CLIENT ( 
    id BIGINT PRIMARY KEY AUTO_INCREMENT, 
    name VARCHAR(255) NOT NULL COMMENT '거래처명', 
    contact VARCHAR(50) COMMENT '연락처', 
    email VARCHAR(100) COMMENT '이메일', 
    address VARCHAR(255) COMMENT '주소', 
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '생성 일시', 
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정 일시'
);


CREATE TABLE EXPEND_CLIENT (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    expend_id BIGINT NOT NULL COMMENT '지출 ID',
    client_id BIGINT NOT NULL COMMENT '거래처 ID',
    amount DECIMAL(18, 2) NOT NULL COMMENT '해당 거래처의 지출 금액',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '생성 일시',
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정 일시',
    CONSTRAINT fk_expend_client_expend FOREIGN KEY (expend_id) REFERENCES EXPEND(id) ON DELETE CASCADE,
    CONSTRAINT fk_expend_client_client FOREIGN KEY (client_id) REFERENCES CLIENT(id) ON DELETE CASCADE
);


CREATE TABLE ACCOUNT_SUBJECT ( 
    id BIGINT PRIMARY KEY AUTO_INCREMENT, 
    account_name VARCHAR(255) NOT NULL COMMENT '회계 계정명', 
    account_code VARCHAR(50) UNIQUE NOT NULL COMMENT '회계 계정 코드', 
    description VARCHAR(255) COMMENT '회계 계정 설명', 
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '생성 일시', 
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정 일시' 
);


CREATE TABLE EXPEND_SUBJECT ( 
    id BIGINT PRIMARY KEY AUTO_INCREMENT, 
    expend_id BIGINT NOT NULL COMMENT '지출 ID', 
    account_subject_id BIGINT NOT NULL COMMENT '회계 계정 ID', 
    subject_name VARCHAR(255) NOT NULL COMMENT '지출 항목명', 
    amount DECIMAL(18, 2) NOT NULL COMMENT '해당 항목의 지출 금액', 
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '생성 일시', 
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정 일시', 
    CONSTRAINT fk_expend_subject_expend FOREIGN KEY (expend_id) REFERENCES EXPEND(id) ON DELETE CASCADE, 
    CONSTRAINT fk_expend_subject_account FOREIGN KEY (account_subject_id) REFERENCES ACCOUNT_SUBJECT(id) ON DELETE CASCADE
);

```
